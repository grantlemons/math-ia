\newcommand{\sumdiv}{3}
\newcommand{\leftbound}{-3.5}
\newcommand{\rightbound}{1}

\newcommand{\graphFunc}{\draw[darkgray, smooth, thick, samples=100, domain=\leftbound:\rightbound] plot(\x,{((\x+(2/3))^3 + 2*(\x)^2)/3}) node[darkgray, align=left, right, yshift=0.5cm, xshift=-3.9cm] {\scriptsize \(f(x)\)};}
\begin{figure}[ht]
    \centering
    \resizebox{.7\textwidth}{!}{
        \begin{tikzpicture}[extended line/.style={shorten >=-#1,shorten <=-#1}]
            \pgfmathparse{int(\sumdiv-1)}
            \foreach \i in {0, ..., \pgfmathresult}
            {
                \pgfmathsetmacro\BLx{\leftbound + (((\rightbound - \leftbound)/\sumdiv)*\i)}
                \pgfmathsetmacro\TRx{\leftbound + (((\rightbound - \leftbound)/\sumdiv)*(\i+1))}
                \pgfmathsetmacro\TRy{((\BLx+(2/3))^3 + 2*(\BLx)^2)/3}
                \coordinate (BL) at (\BLx, 0);
                \coordinate (TR) at (\TRx, \TRy);

                \path [draw = gray, thin, pattern=north west lines, pattern color=gray] (BL) rectangle (TR);
            }

            \draw[black, thick] ({\leftbound-.1}, -0.02) -- ({\rightbound+.1}, -0.02) node[black, align=center, left, yshift=-0.13cm, xshift=-4.3cm] {\scriptsize \(x\)};
            \graphFunc
        \end{tikzpicture}
    }
    \vspace{-.4cm}
    \caption{Rectangle Rule}
    \label{fig:rect_rule_simple}
\end{figure}
